14-Pedir por teclado el día de la semana y escribir si es lectivo o festivo(fin de semana)


package condicionales;

import java.util.Scanner;
public class Ejercicio14 {
    
    public static void main(String[] args) {
         
        Scanner lector=new Scanner(System.in);
        String dia;
        System.out.print(" Introduce un día de la semana: ");
        dia=lector.nextLine();
        dia=dia.toLowerCase();//ponemos toLowerCase para que independientemente de que nosotros pongamos mayúsc
        
        switch(dia){
        
            case "lunes"://debemos poner el nombre de los días de la semana en minúscula para que tenga efecto toLowerCase
                
            case "martes":
                
            case "miercoles":
                          
            case "jueves":
              
            case "viernes":
                System.out.print(dia + " es Lectivo");
                break;
            case "sabado":
                
            case "domingo":
                System.out.print(dia + " es Festivo");  
                break;
                
            default:
                System.out.print("El dato introducido es incorrecto");
                
        }
    }
}
